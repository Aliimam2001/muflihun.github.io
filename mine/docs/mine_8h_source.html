284  template <class Iter>
285  static std::string encode(const Iter& begin, const Iter& end) noexcept
286  {
287  std::string padding;
288  std::stringstream ss;
289  for (auto it = begin; it < end; it += 3) {
290 
291  //
292  // we use example following example for implementation basis
293  // Bits 01100001 01100010 01100011
294  // 24-bit stream: 011000 010110 001001 100011
295  // result indices 24 22 9 35
296  //
297 
298  int c = static_cast<int>(*it & 0xff);
299  ss << static_cast<char>(static_cast<char>(kValidChars[(c >> 2) & 0x3f])); // first 6 bits from first bitset
300  if (it + 1 < end) {
301  int c2 = static_cast<int>(*(it + 1) & 0xff);
302  ss << static_cast<char>(kValidChars[((c << 4) | // remaining 2 bits from first bitset - shift them left to get 4-bit spaces 010000
303  (c2 >> 4) // first 4 bits of second bitset - shift them right to get 2 spaces and bitwise
304  // to add them 000110
305  ) & 0x3f]); // must be within 63 --
306  // 010000
307  // 000110
308  // --|---
309  // 010110
310  // 111111
311  // ---&--
312  // 010110 ==> 22
313  if (it + 2 < end) {
314  int c3 = static_cast<int>(*(it + 2) & 0xff);
315  ss << static_cast<char>(kValidChars[((c2 << 2) | // remaining 4 bits from second bitset - shift them to get 011000
316  (c3 >> 6) // the first 2 bits from third bitset - shift them right to get 000001
317  ) & 0x3f]);
318  // the rest of the explanation is same as above
319  ss << static_cast<char>(kValidChars[c3 & 0x3f]); // all the remaing bits
320  } else {
321  ss << static_cast<char>(kValidChars[(c2 << 2) & 0x3f]); // we have 4 bits left from last byte need space for two 0-bits
322  ss << "=";
323  }
324  } else {
325  ss << static_cast<char>(kValidChars[(c << 4) & 0x3f]); // remaining 2 bits from single byte
326  ss << "==";
327  }
328  }
329  return ss.str() + padding;
330  }

350  template <class Iter>
351  static std::string decode(const Iter& begin, const Iter& end)
352  {
353  //
354  // we use example following example for implementation basis
355  // Bits 01100001 01100010 01100011
356  // 24-bit stream: 011000 010110 001001 100011
357  // result indices 24 22 9 35
358  //
359 
360  auto findPosOf = [](char c) -> int {
361  try {
362  return kDecodeMap.at(static_cast<int>(c & 0xff));
363  } catch (const std::exception& e) {
364  throw e;
365  }
366  };
367 
368  std::stringstream ss;
369  for (auto it = begin; it < end; it += 4) {
370  try {
371  while (iswspace(*it)) {
372  ++it;
373 
374  if (it >= end) {
375  goto result;
376  }
377  }
378  int b0 = findPosOf(*it);
379  if (b0 == kPadding) {
380  throw std::invalid_argument("No data available");
381  }
382  if (b0 == -1) {
383  throw std::invalid_argument("Invalid base64 encoding");
384  }
385 
386  while (iswspace(*(it + 1))) {
387  ++it;
388 
389  if (it >= end) {
390  goto result;
391  }
392  }
393  int b1 = findPosOf(*(it + 1));
394  if (b1 == -1) {
395  throw std::invalid_argument("Invalid base64 encoding");
396  }
397 
398  while (iswspace(*(it + 2))) {
399  ++it;
400 
401  if (it >= end) {
402  goto result;
403  }
404  }
405  int b2 = findPosOf(*(it + 2));
406  if (b2 == -1) {
407  throw std::invalid_argument("Invalid base64 encoding");
408  }
409 
410  while (iswspace(*(it + 3))) {
411  ++it;
412 
413  if (it >= end) {
414  goto result;
415  }
416  }
417  int b3 = findPosOf(*(it + 3));
418  if (b3 == -1) {
419  throw std::invalid_argument("Invalid base64 encoding");
420  }
421 
422  ss << static_cast<byte>(b0 << 2 | // 011000 << 2 ==> 01100000
423  b1 >> 4); // 000001 >> 4 ==> 01100001 ==> 11000001 = 97
424 
425  if (b1 != kPadding) {
426  if (b2 == kPadding) {
427  // second bitset is only 4 bits
428  } else {
429  ss << static_cast<byte>((b1 & ~(1 << 5) & ~(1 << 4)) << 4 | // 010110 ==> 000110 << 4 ==> 1100000
430  // first we clear the bits at pos 4 and 5
431  // then we concat with next bit
432  b2 >> 2); // 001001 >> 2 ==> 00000010 ==> 01100010 = 98
433  if (b3 == kPadding) {
434  // third bitset is only 4 bits
435  } else {
436  ss << static_cast<byte>((b2 & ~(1 << 5) & ~(1 << 4) & ~(1 << 3) & ~(1 << 2)) << 6 | // 001001 ==> 000001 << 6 ==> 01000000
437  // first we clear first 4 bits
438  // then concat with last byte as is
439  b3); // as is
440  }
441  }
442  }
443  } catch (const std::exception& e) {
444  throw std::invalid_argument(std::string("Invalid base64 encoding: " + std::string(e.what())));
445  }
446  }
447 result:
448  return ss.str();
449  }

927  virtual BigInteger modInverse(BigInteger a, BigInteger b) const
928  {
929  BigInteger b0 = b, t, q;
930  BigInteger x0 = 0, x1 = 1;
931  if (b == 1) {
932  return 1;
933  }
934  while (a > 1) {
935  q = a / b;
936  t = b;
937  b = a % b;
938  a = t;
939  t = x0;
940  x0 = x1 - q * x0;
941  x1 = t;
942  }
943  if (x1 < 0) {
944  x1 += b0;
945  }
946  return x1;
947  }

970  virtual BigInteger powerMod(BigInteger b, BigInteger e, BigInteger m) const
971  {
972  BigInteger res = 1;
973  while (e > 0) {
974  if (e % 2 != 0) {
975  res = (b * res) % m;

978  e /= 2;

986  virtual BigInteger power(BigInteger b, BigInteger e) const
987  {
988  BigInteger result = 1;
989  while (e > 0) {
990  if (e % 2 == 1) {
991  // we decrement exponent to make it even
992  e--;
993  // store this multiplication directly to the
994  // result
995  result *= b;
996  // we modify this alg to ignore the next multiplication
997  // if we have already reached 0 (for speed)
998  // here are details and what we changed and how it all works
999  //
1000  // Let's say we have case of 2 ^ 4 [expected answer = 16]
1001  // 2 ^ 4 -- b = 4, e = 2 [result = 1]
1002  // 2 ^ 2 -- b = 16, e = 1 [result = 1]
1003  // 2 ^ 1 -- e = 0 [result = 1 * 16]
1004  //
1005  // here is what we changed here
1006  // now we have result set and we have e set to zero
1007  // doing another b ^= b means b = 16 * 16 = 256 (in our case)
1008  // which is useless so we end here
1009  if (e == 0) {
1010  break;
1011  }
1012  }
1013  e /= 2;
1014  b *= b;
1015  }
1016  return result;
1017  }
