292  static std::size_t expectedDataSize(std::size_t plainDataSize, std::size_t clientIdSize = 16) noexcept
293  {
294  static const int DATA_DELIMITER_LENGTH = sizeof(DATA_DELIMITER);
295 
296  std::size_t dataSize = 32 /* IV */
297  + DATA_DELIMITER_LENGTH
298  + (clientIdSize > 0 ? clientIdSize + DATA_DELIMITER_LENGTH : 0)
299  + expectedBase64Length(expectedAESCipherLength(plainDataSize));
300  unsigned int digits = 0;
301  unsigned int n = static_cast<unsigned int>(dataSize);
302  while (n) {
303  n /= 10;
304  ++digits;
305  };
306  return digits + DATA_DELIMITER_LENGTH + dataSize;
307  }
